create database circle2;

use circle2;

create table circle(radius int(20),dimeter int(20),area float(30),circumference float(30));

drop procedure if exists circle;

delimiter //

create procedure circle()
   -> begin
   ->    declare radius int;
   ->    declare dimeter int;
   ->    declare area float;
   ->    declare circumference float;
   ->    declare pi float default 3.14;
   ->
   ->    set radius:=5;
   ->  
   ->    cloop: LOOP
   ->       set dimeter:= 2*radius;
   ->       set area:= pi*radius*radius;
   ->       set circumference:=2*pi*radius;
   ->    
   ->       select radius,dimeter,area,circumference;
   ->       set radius=radius+1;
   ->       if radius>9 then
   ->         leave cloop;
   ->    end if;
   ->    select radius;
   ->    select area;
   ->    select circumference;
   ->    select dimeter;
   ->    insert into circle(radius,area,dimeter,circumference) values (radius, area, dimeter,circumference);
   ->  END LOOP cloop;
   ->  end;
   ->  //
delimiter ;

call circle();

select * from circle;
